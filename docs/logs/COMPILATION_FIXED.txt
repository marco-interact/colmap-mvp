```tsx
  src={project.thumbnail} 
  alt={project.name}
  className="w-full h-full object-cover"
  onError={(e) => {
    e.currentTarget.style.display = 'none'
    const parent = e.currentTarget.parentElement
    if (parent) {
      parent.innerHTML = '...'
    }
  }}
/>
</div>  ← Correct closing
```
